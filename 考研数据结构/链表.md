#include <iostream>
#include <cstring>
#define MaxSize 50
#define InitSize 100
typedef int Elemtype;
typedef struct Sqlist{
    int data[MaxSize];
    int length;
} Sqlist;

/*typedef struct{

bool ListInsert(Sqlist &L,int i,Elemtype e){
    if(i<1||i>L.length+1)
        return false;
    if(L.length>=MaxSize)
        return false;
    for(int j=L.length;j>=i;j--)
        L.data[j]=L.data[j-1];
    L.data[i-1]=e;
    L.length++;
    return true;
}

bool ListDelete(Sqlist &L,int i,int &e){
    if(i<1||i>L.length+1)
        return false;
    e=L.data[i-1];
    for(int j=i;j<L.length;j++)
        L.data[j-1]=L.data[j];
    L.length--;
    return true;
}

int LocateElem(Sqlist L,int e){
    int i;
    for(i=0;i<L.length;i++)
        if(L.data[i]==e)
            return i+1;
    return 0;
}
//顺序表的初始化
void init_Sqlist(Sqlist &L){
    L.length = 0;

    //下面的L.data的初始化，写不写都行
    memset(L.data,0,sizeof(L.data));
}
//输出整个顺序表
void print(Sqlist L){
    for(int i=0;i<L.length;++i){
        printf("%d ",L.data[i]);
    }
    printf("\n");
}
int main() {

    Sqlist L;
    //顺序表初始化
    init_Sqlist(L);

    for(int i=1;i<=5;i++){
        ListInsert(L,i,i*10);//对形参i而言，i-1才是L.data的下标
    }
    //输出整个顺序表
    print(L);
    //删除,把删掉的值用x带回来
    int x;
    ListDelete(L,1,x);
    printf("The value you deleted is %d.\n",x);
    //输出整个顺序表，看是否删掉了
    print(L);
    //确定某个元素的位置；比如：30
    printf("location is %d.\n",LocateElem(L,30));
}

bool ListInsert(Seqlist &L,int i,Elemtype e){
    //由于是动态数组，当输入的数据范围超过L.MaxSize时，我们扩大L.MaxSize，再进行插入；
    //缺点很明显，我们需要把以前的数组，拷贝到新的数组里面
    if(i<1)
        return false;
    //如果原来的MaxSize不够了
    if(L.length == L.MaxSize){
        //扩大L.MaxSize
        L.MaxSize *= 2;
        Elemtype *data2 = (Elemtype*)malloc(sizeof(Elemtype)*L.MaxSize);
        //空间分配失败返回false
        if(!data2)
            return false;
        //把原来的数据拷贝过来
        //for循环内，习惯用++index(看见大佬都这么写),而不是index++；因为++index效率高
        for(int index=0;index<i;++index){
            data2[index] = L.data[index];
        }
        data2[i-1] = e;
        //小心边界,carefully.

        Elemtype *temp = L.data;//为了释放掉原来的数组，能省则省。
        L.data = data2;
        free(temp);
        L.length++;
        return true;

    for(int j=L.length;j>=i;j--)
